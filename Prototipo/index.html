void preencheVetor(int *vet, int size){
	int i;
	vet = (int*) malloc(sizeof(int) * size);
	for(i=0; i < size; i++){
		vet[i] = i;
	}
	for(i=0; i < size; i++){
		printf("%d ",vet[i]);
	}
	printf("\n");
}
